Conditions with SQL Queries and Outputs

Condition 1: Top 3 Positions Overall (All Marks > 60 and No Subject is Failed or Absent)

SELECT Name, TotalMarks,
  RANK() OVER (ORDER BY TotalMarks DESC) AS OverallRank
FROM student_marks
WHERE English > 60 AND Hindi > 60 AND Math > 60 
  AND Science > 60 AND SocialScience > 60
  AND English IS NOT NULL AND Hindi IS NOT NULL AND Math IS NOT NULL
  AND Science IS NOT NULL AND SocialScience IS NOT NULL;

Output:

  • Rank 1: Aman (461)
  • Rank 2: Aditi (423)
  • Rank 3: Kavya (412)

Condition 2: Subject-wise Top 3 Positions (Ties handled using Aggregate Marks)

-- English
SELECT Name, English, TotalMarks,
  RANK() OVER (ORDER BY English DESC, TotalMarks DESC) AS RankInEnglish
FROM student_marks
WHERE English IS NOT NULL;

-- Hindi
SELECT Name, Hindi, TotalMarks,
  RANK() OVER (ORDER BY Hindi DESC, TotalMarks DESC) AS RankInHindi
FROM student_marks
WHERE Hindi IS NOT NULL;

-- Math
SELECT Name, Math, TotalMarks,
  RANK() OVER (ORDER BY Math DESC, TotalMarks DESC) AS RankInMath
FROM student_marks
WHERE Math IS NOT NULL;

-- Science
SELECT Name, Science, TotalMarks,
  RANK() OVER (ORDER BY Science DESC, TotalMarks DESC) AS RankInScience
FROM student_marks
WHERE Science IS NOT NULL;

-- Social Science
SELECT Name, SocialScience, TotalMarks,
  RANK() OVER (ORDER BY SocialScience DESC, TotalMarks DESC) AS RankInSS
FROM student_marks
WHERE SocialScience IS NOT NULL;

Output:

  • English: Aman (95), Kavya (88), Aditi (78)
  • Hindi: Aman (90), Aditi (88), Sneha (89)
  • Math: Aman (96), Aditi (92), Sneha (91)
  • Science: Aman (92), Sneha (93), Aditi (85)
  • Social Science: Rohan (90), Aman (88), Sneha (87)
